          WHEN END_TIME LIKE 'B+H%'                          THEN ADD_SECONDS(TO_TIMESTAMP(BEGIN_TIME, 'YYYY/MM/DD HH24:MI:SS'), SUBSTR_AFTER(END_TIME, 'B+H') * 3600)
          WHEN END_TIME LIKE 'B+D%'                          THEN ADD_SECONDS(TO_TIMESTAMP(BEGIN_TIME, 'YYYY/MM/DD HH24:MI:SS'), SUBSTR_AFTER(END_TIME, 'B+D') * 86400)
          WHEN END_TIME LIKE 'B+W%'                          THEN ADD_SECONDS(TO_TIMESTAMP(BEGIN_TIME, 'YYYY/MM/DD HH24:MI:SS'), SUBSTR_AFTER(END_TIME, 'B+W') * 86400 * 7)
          WHEN END_TIME =    'MAX'                           THEN TO_TIMESTAMP('9999/12/31 00:00:00', 'YYYY/MM/DD HH24:MI:SS')
          WHEN SUBSTR(END_TIME, 1, 1) NOT IN ('C', 'B', 'M') THEN TO_TIMESTAMP(END_TIME, 'YYYY/MM/DD HH24:MI:SS')
        END END_TIME,
        APP_SERVER,
        MIN_ACT_WPS,
        MIN_ACT_DIA,
        MIN_PRIV_WPS,
        MAX_TRFC_FREE,
        MIN_CPU_USER_PCT,
        MIN_CPU_SYS_PCT,
        MIN_PAGE_OUT_PCT,
        MAX_FREEMEM_PCT,
        MAX_FREEMEM_GB,
        MIN_EXTMEM_GB,
        MIN_HEAPMEM_GB,
        MIN_DIA_QUEUE,
        MIN_UPD_QUEUE,
        MIN_ENQ_QUEUE,
        MIN_ENQ_COUNT,
        MIN_QIN_QUEUE,
        MIN_QOUT_QUEUE,
        MIN_RTT_MS,
        AGGREGATE_BY,
        MAP(TIME_AGGREGATE_BY,
          'NONE',        'YYYY/MM/DD HH24:MI:SS',
          'HOUR',        'YYYY/MM/DD HH24',
          'DAY',         'YYYY/MM/DD (DY)',
          'HOUR_OF_DAY', 'HH24',
          TIME_AGGREGATE_BY ) TIME_AGGREGATE_BY,
        ORDER_BY
      FROM
      ( SELECT                          /* Modification section */
          'C-H1' BEGIN_TIME,                  /* YYYY/MM/DD HH24:MI:SS timestamp, C, C-S<seconds>, C-M<minutes>, C-H<hours>, C-D<days>, C-W<weeks>, E-S<seconds>, E-M<minutes>, E-H<hours>, E-D<days>, E-W<weeks>, MIN */
          'C' END_TIME,                    /* YYYY/MM/DD HH24:MI:SS timestamp, C, C-S<seconds>, C-M<minutes>, C-H<hours>, C-D<days>, C-W<weeks>, B+S<seconds>, B+M<minutes>, B+H<hours>, B+D<days>, B+W<weeks>, MAX */
          '%' APP_SERVER,
          -1 MIN_ACT_WPS,
          -1 MIN_ACT_DIA,
          -1 MIN_PRIV_WPS,
          -1 MIN_CPU_USER_PCT,
          -1 MIN_CPU_SYS_PCT,
          -1 MIN_PAGE_OUT_PCT,
          -1 MAX_FREEMEM_PCT,
          -1 MAX_FREEMEM_GB,
          -1 MIN_EXTMEM_GB,
          -1 MIN_HEAPMEM_GB,
          -1 MIN_DIA_QUEUE,
          -1 MIN_UPD_QUEUE,
          -1 MIN_ENQ_QUEUE,
          -1 MIN_ENQ_COUNT,
          -1 MIN_QIN_QUEUE,
          -1 MIN_QOUT_QUEUE,
          -1 MAX_TRFC_FREE,
          -1 MIN_RTT_MS,
          'NONE' AGGREGATE_BY,         /* TIME, SERVER or comma separated combinations, NONE for no aggregation */
          'NONE' TIME_AGGREGATE_BY,    /* HOUR, DAY, HOUR_OF_DAY or database time pattern, TS<seconds> for time slice, NONE for no aggregation */
          'TIME' ORDER_BY              /* TIME, SAMPLES, SERVER, ACT_WPS, ACT_DIA, PRIV_WPS, TRFC_FREE, CPU_USER, CPU_SYS, PAGE_OUT, FREEMEMPCT, FREEMEMGB, EXTMEM, HEAPMEM, DIA_QUEUE, UPD_QUEUE, ENQ_QUEUE, QIN_QUEUE, QOUT_QUEUE */
        FROM
          DUMMY
      )
    ) BI,
    ( SELECT
        TO_TIMESTAMP(DATUM || TIME, 'YYYYMMDDHH24MISS') START_TIME,
        SERVER APP_SERVER,
        ACT_WPS,
        ACT_DIA,
        PRIVWPNO PRIV_WPS,
        TRFC_FREE,
        USR_TOTAL CPU_USER_PCT,
        SYS_TOTAL CPU_SYS_PCT,
        PAGE_OUT_PERC PAGE_OUT_PCT,
        FREE_MEM_PERC FREEMEM_PCT,
        FREE_MEM_MB_INC_FS / 1024 FREEMEM_GB,
        EMALLOC / 1024 EXTMEM_GB,
        HEAPSUMKB / 1024 / 1024 HEAPMEM_GB,
        DIAQ DIA_QUEUE,
        UPDQ UPD_QUEUE,
        ENQQ ENQ_QUEUE,
        SM12_CNT ENQ_COUNT,
        QINLENGTH QIN_QUEUE,
        QOUTLENGTH QOUT_QUEUE,
        USERS,
        SESSIONS,
        NETRTT / 1000 RTT_MS
      FROM
        "/SDF/SMON_HEADER"
    ) S
    WHERE
      S.START_TIME BETWEEN BI.BEGIN_TIME AND BI.END_TIME AND
      S.APP_SERVER LIKE BI.APP_SERVER
    GROUP BY
      S.START_TIME,
      CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'SERVER') != 0 THEN S.APP_SERVER ELSE MAP(BI.APP_SERVER, '%', 'any', BI.APP_SERVER) END,
      BI.APP_SERVER,
      BI.MIN_ACT_WPS,
      BI.MIN_ACT_DIA,
      BI.MIN_PRIV_WPS,
      BI.MAX_TRFC_FREE,
      BI.MIN_CPU_USER_PCT,
      BI.MIN_CPU_SYS_PCT,
      BI.MIN_PAGE_OUT_PCT,
      BI.MAX_FREEMEM_PCT,
      BI.MAX_FREEMEM_GB,
      BI.MIN_EXTMEM_GB,
      BI.MIN_HEAPMEM_GB,
      BI.MIN_DIA_QUEUE,
      BI.MIN_UPD_QUEUE,
      BI.MIN_ENQ_QUEUE,
      BI.MIN_ENQ_COUNT,
      BI.MIN_QIN_QUEUE,
      BI.MIN_QOUT_QUEUE,
      BI.MIN_RTT_MS,
      BI.AGGREGATE_BY,
      BI.TIME_AGGREGATE_BY,
      BI.ORDER_BY
  )
  GROUP BY
    CASE
      WHEN AGGREGATE_BY = 'NONE' OR INSTR(AGGREGATE_BY, 'TIME') != 0 THEN 
        CASE 
          WHEN TIME_AGGREGATE_BY LIKE 'TS%' THEN
            TO_VARCHAR(ADD_SECONDS(TO_TIMESTAMP('2014/01/01 00:00:00', 'YYYY/MM/DD HH24:MI:SS'), FLOOR(SECONDS_BETWEEN(TO_TIMESTAMP('2014/01/01 00:00:00', 
            'YYYY/MM/DD HH24:MI:SS'), START_TIME) / SUBSTR(TIME_AGGREGATE_BY, 3)) * SUBSTR(TIME_AGGREGATE_BY, 3)), 'YYYY/MM/DD HH24:MI:SS')
          ELSE TO_VARCHAR(START_TIME, TIME_AGGREGATE_BY)
        END
      ELSE 'any' 
    END,
    APP_SERVER,
    MIN_ACT_WPS,
    MIN_ACT_DIA,
    MIN_PRIV_WPS,
    MAX_TRFC_FREE,
    MIN_CPU_USER_PCT,
    MIN_CPU_SYS_PCT,
    MIN_PAGE_OUT_PCT,
    MAX_FREEMEM_PCT,
    MAX_FREEMEM_GB,
    MIN_EXTMEM_GB,
    MIN_HEAPMEM_GB,
    MIN_DIA_QUEUE,
    MIN_UPD_QUEUE,
    MIN_ENQ_QUEUE,
    MIN_ENQ_COUNT,
    MIN_QIN_QUEUE,
    MIN_QOUT_QUEUE,
    MIN_RTT_MS,
    ORDER_BY
)
WHERE
  ( MIN_ACT_WPS      = -1 OR ACT_WPS      >= MIN_ACT_WPS      ) AND
  ( MIN_ACT_DIA      = -1 OR ACT_DIA      >= MIN_ACT_DIA      ) AND
  ( MIN_PRIV_WPS     = -1 OR PRIV_WPS     >= MIN_PRIV_WPS     ) AND
  ( MAX_TRFC_FREE    = -1 OR TRFC_FREE    <= MAX_TRFC_FREE    ) AND
  ( MIN_CPU_USER_PCT = -1 OR CPU_USER_PCT >= MIN_CPU_USER_PCT ) AND
  ( MIN_CPU_SYS_PCT  = -1 OR CPU_SYS_PCT  >= MIN_CPU_SYS_PCT  ) AND
  ( MIN_PAGE_OUT_PCT = -1 OR PAGE_OUT_PCT >= MIN_PAGE_OUT_PCT ) AND
  ( MAX_FREEMEM_PCT  = -1 OR FREEMEM_PCT  <= MAX_FREEMEM_PCT  ) AND
  ( MAX_FREEMEM_GB   = -1 OR FREEMEM_GB   <= MAX_FREEMEM_GB   ) AND
  ( MIN_EXTMEM_GB    = -1 OR EXTMEM_GB    >= MIN_EXTMEM_GB    ) AND
  ( MIN_HEAPMEM_GB   = -1 OR HEAPMEM_GB   >= MIN_HEAPMEM_GB   ) AND
  ( MIN_DIA_QUEUE    = -1 OR DIA_QUEUE    >= MIN_DIA_QUEUE    ) AND
  ( MIN_UPD_QUEUE    = -1 OR UPD_QUEUE    >= MIN_UPD_QUEUE    ) AND
  ( MIN_ENQ_QUEUE    = -1 OR ENQ_QUEUE    >= MIN_ENQ_QUEUE    ) AND
  ( MIN_ENQ_COUNT    = -1 OR ENQ_COUNT    >= MIN_ENQ_COUNT    ) AND
  ( MIN_QIN_QUEUE    = -1 OR QIN_QUEUE    >= MIN_QIN_QUEUE    ) AND
  ( MIN_QOUT_QUEUE   = -1 OR QOUT_QUEUE   >= MIN_QOUT_QUEUE   ) AND
  ( MIN_RTT_MS       = -1 OR RTT_MS       >= MIN_RTT_MS       )
ORDER BY
  MAP(ORDER_BY, 'SAMPLES', CNT) DESC,
  MAP(ORDER_BY, 'TIME', START_TIME) DESC,
  MAP(ORDER_BY, 'ACT_WPS', ACT_WPS, 'ACT_DIA', ACT_DIA, PRIV_WPS, 'PRIV_WPS', 'CPU_USER', CPU_USER_PCT, 'CPU_SYS', CPU_SYS_PCT, 'PAGE_OUT', PAGE_OUT_PCT,
    'EXTMEM', EXTMEM_GB, 'HEAPMEM', HEAPMEM_GB, 'DIA_QUEUE', DIA_QUEUE, 'UPD_QUEUE', UPD_QUEUE, 'ENQ_QUEUE', ENQ_QUEUE, 'QIN_QUEUE', QIN_QUEUE, 'QOUT_QUEUE', QOUT_QUEUE) DESC,
  MAP(ORDER_BY, 'SERVER', APP_SERVER, 'TRFC_FREE', TRFC_FREE, 'FREEMEMGB', FREEMEM_GB, 'FREEMEMPCT', FREEMEM_PCT),
  START_TIME DESC,
  APP_SERVER
