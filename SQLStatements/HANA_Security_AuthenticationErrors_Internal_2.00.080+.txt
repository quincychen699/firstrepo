  'TIME'          --> Aggregation by time

|2024/11/24 12:16:31|ABCDE      |c11adm  |A01: User ABCDE does not exist      |CLIENT CONNECT|PBKDF2     | 354840|sapabap    |12.123.123.123|47585     |SQLDBC     |2.12.22.1649876447               |
|2024/11/24 12:16:31|ABCDE      |c11adm  |A01: User ABCDE does not exist      |CLIENT CONNECT|PBKDF2     | 354839|sapabap    |12.123.123.123|47567     |SQLDBC     |2.12.22.1649876447               |

  ERROR_TIME,
  HOST,
  LPAD(PORT, 5) PORT,
  LPAD(CNT, 5) "COUNT",
  DB_USER,
  APP_USER,
  ERROR,
  AUTH_FOR,
  AUTH_METHOD,
  LPAD(CONN_ID, 7) CONN_ID,
  CLIENT_HOST,
  CLIENT_IP,
  CLIENT_PID,
  CLIENT_TYPE,
  CLIENT_VERSION
FROM
( SELECT
    CASE 
      WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'TIME') != 0 THEN 
        CASE 
          WHEN BI.TIME_AGGREGATE_BY LIKE 'TS%' THEN
          TO_VARCHAR(ADD_SECONDS(TO_TIMESTAMP('2014/01/01 00:00:00', 'YYYY/MM/DD HH24:MI:SS'), FLOOR(SECONDS_BETWEEN(TO_TIMESTAMP('2014/01/01 00:00:00', 
            'YYYY/MM/DD HH24:MI:SS'), CASE BI.TIMEZONE WHEN 'UTC' THEN ADD_SECONDS(A.TIMESTAMP, SECONDS_BETWEEN(CURRENT_TIMESTAMP, CURRENT_UTCTIMESTAMP)) ELSE A.TIMESTAMP END) / SUBSTR(BI.TIME_AGGREGATE_BY, 3)) * SUBSTR(BI.TIME_AGGREGATE_BY, 3)), 'YYYY/MM/DD HH24:MI:SS')
          ELSE TO_VARCHAR(CASE BI.TIMEZONE WHEN 'UTC' THEN ADD_SECONDS(A.TIMESTAMP, SECONDS_BETWEEN(CURRENT_TIMESTAMP, CURRENT_UTCTIMESTAMP)) ELSE A.TIMESTAMP END, BI.TIME_AGGREGATE_BY)
        END
      ELSE 'any' 
    END ERROR_TIME,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'HOST')           != 0 THEN A.HOST                      ELSE MAP(BI.HOST,           '%', 'any', BI.HOST)                   END HOST,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'PORT')           != 0 THEN TO_VARCHAR(A.PORT)          ELSE MAP(BI.PORT,           '%', 'any', BI.PORT)                   END PORT,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'CONN_ID')        != 0 THEN TO_VARCHAR(A.CONNECTION_ID) ELSE MAP(BI.CONN_ID,         -1, 'any', TO_VARCHAR(BI.CONN_ID))    END CONN_ID,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'AUTH_FOR')       != 0 THEN A.AUTHENTICATION_FOR        ELSE MAP(BI.AUTH_FOR,       '%', 'any', BI.AUTH_FOR)               END AUTH_FOR,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'AUTH_METHOD')    != 0 THEN A.AUTHENTICATION_METHOD     ELSE MAP(BI.AUTH_METHOD,    '%', 'any', BI.AUTH_METHOD)            END AUTH_METHOD,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'ERROR')          != 0 THEN A.INTERNAL_ERROR_CODE || ':' || CHAR(32) || A.REASON      
                                                                                                                         ELSE MAP(BI.ERROR,          '%', 'any', BI.ERROR)                  END ERROR,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'DB_USER')        != 0 THEN A.USER_NAME                 ELSE MAP(BI.DB_USER,        '%', 'any', BI.DB_USER)                END DB_USER,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'APP_USER')       != 0 THEN A.APPLICATION_USER_NAME     ELSE MAP(BI.APP_USER,       '%', 'any', BI.APP_USER)               END APP_USER,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'CLIENT_HOST')    != 0 THEN A.CLIENT_HOST               ELSE MAP(BI.CLIENT_HOST,    '%', 'any', BI.CLIENT_HOST)            END CLIENT_HOST,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'CLIENT_HOST')    != 0 THEN A.CLIENT_IP                 ELSE MAP(BI.CLIENT_IP,      '%', 'any', BI.CLIENT_IP)              END CLIENT_IP,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'CLIENT_PID')     != 0 THEN TO_VARCHAR(A.CLIENT_PID)    ELSE MAP(BI.CLIENT_PID,      -1, 'any', TO_VARCHAR(BI.CLIENT_PID)) END CLIENT_PID,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'CLIENT_TYPE')    != 0 THEN A.CLIENT_TYPE               ELSE MAP(BI.CLIENT_TYPE,    '%', 'any', BI.CLIENT_TYPE)            END CLIENT_TYPE,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'CLIENT_VERSION') != 0 THEN A.CLIENT_VERSION            ELSE MAP(BI.CLIENT_VERSION, '%', 'any', BI.CLIENT_VERSION)         END CLIENT_VERSION,
    COUNT(*) CNT,
    BI.ORDER_BY
  FROM
  ( SELECT
      CASE
        WHEN BEGIN_TIME =    'C'                             THEN CURRENT_TIMESTAMP
        WHEN BEGIN_TIME LIKE 'C-S%'                          THEN ADD_SECONDS(CURRENT_TIMESTAMP, -SUBSTR_AFTER(BEGIN_TIME, 'C-S'))
        WHEN BEGIN_TIME LIKE 'C-M%'                          THEN ADD_SECONDS(CURRENT_TIMESTAMP, -SUBSTR_AFTER(BEGIN_TIME, 'C-M') * 60)
        WHEN BEGIN_TIME LIKE 'C-H%'                          THEN ADD_SECONDS(CURRENT_TIMESTAMP, -SUBSTR_AFTER(BEGIN_TIME, 'C-H') * 3600)
        WHEN BEGIN_TIME LIKE 'C-D%'                          THEN ADD_SECONDS(CURRENT_TIMESTAMP, -SUBSTR_AFTER(BEGIN_TIME, 'C-D') * 86400)
        WHEN BEGIN_TIME LIKE 'C-W%'                          THEN ADD_SECONDS(CURRENT_TIMESTAMP, -SUBSTR_AFTER(BEGIN_TIME, 'C-W') * 86400 * 7)
        WHEN BEGIN_TIME LIKE 'E-S%'                          THEN ADD_SECONDS(TO_TIMESTAMP(END_TIME, 'YYYY/MM/DD HH24:MI:SS'), -SUBSTR_AFTER(BEGIN_TIME, 'E-S'))
        WHEN BEGIN_TIME LIKE 'E-M%'                          THEN ADD_SECONDS(TO_TIMESTAMP(END_TIME, 'YYYY/MM/DD HH24:MI:SS'), -SUBSTR_AFTER(BEGIN_TIME, 'E-M') * 60)
        WHEN BEGIN_TIME LIKE 'E-H%'                          THEN ADD_SECONDS(TO_TIMESTAMP(END_TIME, 'YYYY/MM/DD HH24:MI:SS'), -SUBSTR_AFTER(BEGIN_TIME, 'E-H') * 3600)
        WHEN BEGIN_TIME LIKE 'E-D%'                          THEN ADD_SECONDS(TO_TIMESTAMP(END_TIME, 'YYYY/MM/DD HH24:MI:SS'), -SUBSTR_AFTER(BEGIN_TIME, 'E-D') * 86400)
        WHEN BEGIN_TIME LIKE 'E-W%'                          THEN ADD_SECONDS(TO_TIMESTAMP(END_TIME, 'YYYY/MM/DD HH24:MI:SS'), -SUBSTR_AFTER(BEGIN_TIME, 'E-W') * 86400 * 7)
        WHEN BEGIN_TIME =    'MIN'                           THEN TO_TIMESTAMP('1000/01/01 00:00:00', 'YYYY/MM/DD HH24:MI:SS')
        WHEN SUBSTR(BEGIN_TIME, 1, 1) NOT IN ('C', 'E', 'M') THEN TO_TIMESTAMP(BEGIN_TIME, 'YYYY/MM/DD HH24:MI:SS')
      END BEGIN_TIME,
      CASE
        WHEN END_TIME =    'C'                             THEN CURRENT_TIMESTAMP
        WHEN END_TIME LIKE 'C-S%'                          THEN ADD_SECONDS(CURRENT_TIMESTAMP, -SUBSTR_AFTER(END_TIME, 'C-S'))
        WHEN END_TIME LIKE 'C-M%'                          THEN ADD_SECONDS(CURRENT_TIMESTAMP, -SUBSTR_AFTER(END_TIME, 'C-M') * 60)
        WHEN END_TIME LIKE 'C-H%'                          THEN ADD_SECONDS(CURRENT_TIMESTAMP, -SUBSTR_AFTER(END_TIME, 'C-H') * 3600)
        WHEN END_TIME LIKE 'C-D%'                          THEN ADD_SECONDS(CURRENT_TIMESTAMP, -SUBSTR_AFTER(END_TIME, 'C-D') * 86400)
        WHEN END_TIME LIKE 'C-W%'                          THEN ADD_SECONDS(CURRENT_TIMESTAMP, -SUBSTR_AFTER(END_TIME, 'C-W') * 86400 * 7)
        WHEN END_TIME LIKE 'B+S%'                          THEN ADD_SECONDS(TO_TIMESTAMP(BEGIN_TIME, 'YYYY/MM/DD HH24:MI:SS'), SUBSTR_AFTER(END_TIME, 'B+S'))
        WHEN END_TIME LIKE 'B+M%'                          THEN ADD_SECONDS(TO_TIMESTAMP(BEGIN_TIME, 'YYYY/MM/DD HH24:MI:SS'), SUBSTR_AFTER(END_TIME, 'B+M') * 60)
        WHEN END_TIME LIKE 'B+H%'                          THEN ADD_SECONDS(TO_TIMESTAMP(BEGIN_TIME, 'YYYY/MM/DD HH24:MI:SS'), SUBSTR_AFTER(END_TIME, 'B+H') * 3600)
        WHEN END_TIME LIKE 'B+D%'                          THEN ADD_SECONDS(TO_TIMESTAMP(BEGIN_TIME, 'YYYY/MM/DD HH24:MI:SS'), SUBSTR_AFTER(END_TIME, 'B+D') * 86400)
        WHEN END_TIME LIKE 'B+W%'                          THEN ADD_SECONDS(TO_TIMESTAMP(BEGIN_TIME, 'YYYY/MM/DD HH24:MI:SS'), SUBSTR_AFTER(END_TIME, 'B+W') * 86400 * 7)
        WHEN END_TIME =    'MAX'                           THEN TO_TIMESTAMP('9999/12/31 00:00:00', 'YYYY/MM/DD HH24:MI:SS')
        WHEN SUBSTR(END_TIME, 1, 1) NOT IN ('C', 'B', 'M') THEN TO_TIMESTAMP(END_TIME, 'YYYY/MM/DD HH24:MI:SS')
      END END_TIME,
      TIMEZONE,
      HOST,
      PORT,
      CONN_ID,
      AUTH_FOR,
      AUTH_METHOD,
      ERROR,
      DB_USER,
      APP_USER,
      CLIENT_HOST,
      CLIENT_IP,
      CLIENT_PID,
      CLIENT_TYPE,
      CLIENT_VERSION,
      AGGREGATE_BY,
      MAP(TIME_AGGREGATE_BY,
        'NONE',        'YYYY/MM/DD HH24:MI:SS',
        'HOUR',        'YYYY/MM/DD HH24',
        'DAY',         'YYYY/MM/DD (DY)',
        'HOUR_OF_DAY', 'HH24',
        TIME_AGGREGATE_BY ) TIME_AGGREGATE_BY,
      ORDER_BY
    FROM
    ( SELECT                                                         /* Modification section */
        '1000/10/18 07:58:00' BEGIN_TIME,                  /* YYYY/MM/DD HH24:MI:SS timestamp, C, C-S<seconds>, C-M<minutes>, C-H<hours>, C-D<days>, C-W<weeks>, E-S<seconds>, E-M<minutes>, E-H<hours>, E-D<days>, E-W<weeks>, MIN */
        '9999/10/18 08:05:00' END_TIME,                    /* YYYY/MM/DD HH24:MI:SS timestamp, C, C-S<seconds>, C-M<minutes>, C-H<hours>, C-D<days>, C-W<weeks>, B+S<seconds>, B+M<minutes>, B+H<hours>, B+D<days>, B+W<weeks>, MAX */
        'SERVER' TIMEZONE,                              /* SERVER, UTC */
        '%' HOST,
        '%' PORT,
         -1 CONN_ID,
        '%' AUTH_FOR,
        '%' AUTH_METHOD,
        '%' ERROR,
        '%' DB_USER,
        '%' APP_USER,
        '%' CLIENT_HOST,
        '%' CLIENT_IP,
         -1 CLIENT_PID,
        '%' CLIENT_TYPE,
        '%' CLIENT_VERSION,
        'NONE' AGGREGATE_BY,          /* TIME, HOST, PORT, CONN_ID, AUTH_FOR, AUTH_METHOD, ERROR, DB_USER, APP_USER, CLIENT_HOST, CLIENT_PID, CLIENT_TYPE, CLIENT_VERSION or comma separated combinations, NONE for no aggregation */
        'NONE' TIME_AGGREGATE_BY,    /* HOUR, DAY, HOUR_OF_DAY or database time pattern, TS<seconds> for time slice, NONE for no aggregation */
        'TIME' ORDER_BY               /* TIME, COUNT, ERROR, USER */   
      FROM
        DUMMY
    )
  ) BI,
    AUTHENTICATION_ERROR_DETAILS A
  WHERE
    A.HOST LIKE BI.HOST AND  
    TO_VARCHAR(A.PORT) LIKE BI.PORT AND
    CASE BI.TIMEZONE WHEN 'UTC' THEN ADD_SECONDS(A.TIMESTAMP, SECONDS_BETWEEN(CURRENT_TIMESTAMP, CURRENT_UTCTIMESTAMP)) ELSE A.TIMESTAMP END BETWEEN BI.BEGIN_TIME AND BI.END_TIME AND
    ( BI.CONN_ID = -1 OR A.CONNECTION_ID = BI.CONN_ID ) AND
    A.AUTHENTICATION_FOR LIKE BI.AUTH_FOR AND
    A.AUTHENTICATION_METHOD LIKE BI.AUTH_METHOD AND
    A.INTERNAL_ERROR_CODE || ':' || CHAR(32) || A.REASON LIKE BI.ERROR AND
    A.USER_NAME LIKE BI.DB_USER AND
    A.APPLICATION_USER_NAME LIKE BI.APP_USER AND
    A.CLIENT_HOST LIKE BI.CLIENT_HOST AND
    A.CLIENT_IP LIKE BI.CLIENT_IP AND
    ( BI.CLIENT_PID = -1 OR A.CLIENT_PID = BI.CLIENT_PID ) AND
    A.CLIENT_TYPE LIKE BI.CLIENT_TYPE AND
    A.CLIENT_VERSION LIKE BI.CLIENT_VERSION
  GROUP BY
    CASE 
      WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'TIME') != 0 THEN 
        CASE 
          WHEN BI.TIME_AGGREGATE_BY LIKE 'TS%' THEN
            TO_VARCHAR(ADD_SECONDS(TO_TIMESTAMP('2014/01/01 00:00:00', 'YYYY/MM/DD HH24:MI:SS'), FLOOR(SECONDS_BETWEEN(TO_TIMESTAMP('2014/01/01 00:00:00', 
            'YYYY/MM/DD HH24:MI:SS'), CASE BI.TIMEZONE WHEN 'UTC' THEN ADD_SECONDS(A.TIMESTAMP, SECONDS_BETWEEN(CURRENT_TIMESTAMP, CURRENT_UTCTIMESTAMP)) ELSE A.TIMESTAMP END) / SUBSTR(BI.TIME_AGGREGATE_BY, 3)) * SUBSTR(BI.TIME_AGGREGATE_BY, 3)), 'YYYY/MM/DD HH24:MI:SS')
          ELSE TO_VARCHAR(CASE BI.TIMEZONE WHEN 'UTC' THEN ADD_SECONDS(A.TIMESTAMP, SECONDS_BETWEEN(CURRENT_TIMESTAMP, CURRENT_UTCTIMESTAMP)) ELSE A.TIMESTAMP END, BI.TIME_AGGREGATE_BY)
        END
      ELSE 'any' 
    END,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'HOST')           != 0 THEN A.HOST                      ELSE MAP(BI.HOST,           '%', 'any', BI.HOST)                   END,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'PORT')           != 0 THEN TO_VARCHAR(A.PORT)          ELSE MAP(BI.PORT,           '%', 'any', BI.PORT)                   END,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'CONN_ID')        != 0 THEN TO_VARCHAR(A.CONNECTION_ID) ELSE MAP(BI.CONN_ID,         -1, 'any', TO_VARCHAR(BI.CONN_ID))    END,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'AUTH_FOR')       != 0 THEN A.AUTHENTICATION_FOR        ELSE MAP(BI.AUTH_FOR,       '%', 'any', BI.AUTH_FOR)               END,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'AUTH_METHOD')    != 0 THEN A.AUTHENTICATION_METHOD     ELSE MAP(BI.AUTH_METHOD,    '%', 'any', BI.AUTH_METHOD)            END,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'ERROR')          != 0 THEN A.INTERNAL_ERROR_CODE || ':' || CHAR(32) || A.REASON      
                                                                                                                         ELSE MAP(BI.ERROR,          '%', 'any', BI.ERROR)                  END,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'DB_USER')        != 0 THEN A.USER_NAME                 ELSE MAP(BI.DB_USER,        '%', 'any', BI.DB_USER)                END,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'APP_USER')       != 0 THEN A.APPLICATION_USER_NAME     ELSE MAP(BI.APP_USER,       '%', 'any', BI.APP_USER)               END,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'CLIENT_HOST')    != 0 THEN A.CLIENT_HOST               ELSE MAP(BI.CLIENT_HOST,    '%', 'any', BI.CLIENT_HOST)            END,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'CLIENT_HOST')    != 0 THEN A.CLIENT_IP                 ELSE MAP(BI.CLIENT_IP,      '%', 'any', BI.CLIENT_IP)              END,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'CLIENT_PID')     != 0 THEN TO_VARCHAR(A.CLIENT_PID)    ELSE MAP(BI.CLIENT_PID,      -1, 'any', TO_VARCHAR(BI.CLIENT_PID)) END,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'CLIENT_TYPE')    != 0 THEN A.CLIENT_TYPE               ELSE MAP(BI.CLIENT_TYPE,    '%', 'any', BI.CLIENT_TYPE)            END,
    CASE WHEN BI.AGGREGATE_BY = 'NONE' OR INSTR(BI.AGGREGATE_BY, 'CLIENT_VERSION') != 0 THEN A.CLIENT_VERSION            ELSE MAP(BI.CLIENT_VERSION, '%', 'any', BI.CLIENT_VERSION)         END,
    BI.ORDER_BY
)
ORDER BY
  MAP(ORDER_BY, 'TIME', ERROR_TIME) DESC,
  MAP(ORDER_BY, 'COUNT', CNT) DESC,
  MAP(ORDER_BY, 'ERROR', ERROR, 'USER', DB_USER || APP_USER) DESC,
  ERROR_TIME DESC,
  CNT DESC,
  ERROR,
  DB_USER,
  APP_USER
